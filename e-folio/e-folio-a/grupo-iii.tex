\begin{equation}
	\begin{cases}
		ax + cy = 1\\
		-ax - 2y -az = -3\\
		2y + 2az = 2c
	\end{cases}
\end{equation}

\begin{align*}
	\intertext{O sistema é equivalente a}
	\begin{bmatrix*}[r]
		a  & c  & 0\\
		-a & -2 & -a\\
		0  & 2  & 2a
	\end{bmatrix*}
	&\begin{bmatrix*}[r]
		x\\
		y\\
		z
	\end{bmatrix*}
	=
	\begin{bmatrix*}[r]
		1\\
		-3\\
		2c
	\end{bmatrix*}\\
	\det
	\begin{bmatrix*}[r]
		a  & c  & 0\\
		-a & -2 & -a\\
		0  & 2  & 2a
	\end{bmatrix*}
	&=
	a \cdot \det
	\begin{bmatrix*}[r]
		 -2 & -a\\
		 2  & 2a
	\end{bmatrix*}
	- c \cdot \det
	\begin{bmatrix*}[r]
		-a & -a\\
		0  & 2a
	\end{bmatrix*}\\
	&= a \cdot [-4a - (-2a)]
	- c \cdot (-2a^2)\\
	&= -2a^2 + c2a^2 = 2a^2(-1 + c)
	\intertext{Para o sistema ter solução, o determinante calculado tem de
	diferir de 0, portanto}
	2a^2(-1 + c) &= 0
	\iff a = 0 \lor c = 1
\end{align*}
